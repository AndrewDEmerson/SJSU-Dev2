333  void Initialize() override
334  {
335  LPC_SC_TypeDef * sys = system_controller;
336  auto config = clock_configuration_;
337 
338  units::frequency::hertz_t system_clock = 0_Hz;
339  units::frequency::hertz_t pll0 = 0_Hz;
340  units::frequency::hertz_t pll1 = 0_Hz;
341 
342  // Reset all of the cached clock frequency values
343  units::frequency::hertz_t cpu = 0_Hz;
344  units::frequency::hertz_t usb = 0_Hz;
345  units::frequency::hertz_t spifi = 0_Hz;
346 
347  // =========================================================================
348  // Step 1. Select IRC as clock source for everything.
349  // Make sure PLLs are not clock sources for everything.
350  // =========================================================================
351  // Set CPU clock to system clock
352  sys->CCLKSEL =
353  bit::Insert(sys->CCLKSEL, Value(CpuClockSelect::kSystemClock),
354  CpuClockRegister::kSelect);
355  // Set USB clock to system clock
356  sys->USBCLKSEL =
357  bit::Insert(sys->USBCLKSEL, Value(UsbClockSelect::kSystemClock),
358  UsbClockRegister::kSelect);
359  // Set SPIFI clock to system clock
360  sys->SPIFISEL =
361  bit::Insert(sys->SPIFISEL, Value(SpifiClockSelect::kSystemClock),
362  SpiFiClockRegister::kSelect);
363 
364  // Set the clock source to IRC and not external oscillator. The next phase
365  // disables that clock source, which will stop the system if this is not
366  // switched.
367  sys->CLKSRCSEL = Value(OscillatorSource::kIrc);
368 
369  // =========================================================================
370  // Step 2. Disable PLLs
371  // =========================================================================
372  // NOTE: The only bit in this register that is used is bit 0 which indicates
373  // enabled or disabled status, thus a single assignment is needed.
374  sys->PLL0CON = 0;
375  sys->PLL1CON = 0;
376  // Disabling external oscillator if it is not going to be used
377  system_controller->SCS =
378  bit::Clear(system_controller->SCS, OscillatorRegister::kExternalEnable);
379 
380  // =========================================================================
381  // Step 3. Select oscillator source for System Clock and Main PLL
382  // =========================================================================
383  // Enable the external oscillator if we are using it, which would be the
384  // case if the alternative PLL is enabled or external oscillator is
385  // selected.
386  if (config.system_oscillator == OscillatorSource::kExternal ||
387  config.pll[1].enabled)
388  {
389  EnableExternalOscillator();
390  }
391 
392  sys->CLKSRCSEL = Value(config.system_oscillator);
393 
394  switch (config.system_oscillator)
395  {
396  case OscillatorSource::kIrc: system_clock = kDefaultIRCFrequency; break;
397  case OscillatorSource::kExternal:
398  system_clock = config.external_oscillator_frequency;
399  break;
400  }
401 
402  // =========================================================================
403  // Step 4. Configure PLLs
404  // =========================================================================
405  pll0 = SetupPll(&sys->PLL0CON, &sys->PLL0CFG, &sys->PLL0FEED,
406  &sys->PLL0STAT, 0);
407 
408  pll1 = SetupPll(&sys->PLL1CON, &sys->PLL1CFG, &sys->PLL1FEED,
409  &sys->PLL1STAT, 1);
410 
411  // =========================================================================
412  // Step 5. Set clock sources for each clock
413  // =========================================================================
414  // Set CPU clock the source defined in the configuration
415  sys->CCLKSEL = bit::Insert(sys->CCLKSEL, Value(config.cpu.clock),
416  CpuClockRegister::kSelect);
417 
418  // Set USB clock the source defined in the configuration
419  sys->USBCLKSEL = bit::Insert(sys->USBCLKSEL, Value(config.usb.clock),
420  UsbClockRegister::kSelect);
421 
422  // Set SPIFI clock the source defined in the configuration
423  sys->SPIFISEL = bit::Insert(sys->SPIFISEL, Value(config.spifi.clock),
424  SpiFiClockRegister::kSelect);
425 
426  switch (config.cpu.clock)
427  {
428  case CpuClockSelect::kSystemClock: cpu = system_clock; break;
429  case CpuClockSelect::kPll0: cpu = pll0; break;
430  }
431 
432  switch (config.usb.clock)
433  {
434  case UsbClockSelect::kSystemClock: usb = system_clock; break;
435  case UsbClockSelect::kPll0: usb = pll0; break;
436  case UsbClockSelect::kPll1: usb = pll1; break;
437  }
438 
439  switch (config.spifi.clock)
440  {
441  case SpifiClockSelect::kSystemClock: spifi = system_clock; break;
442  case SpifiClockSelect::kPll0: spifi = pll0; break;
443  case SpifiClockSelect::kPll1: spifi = pll1; break;
444  }
445 
446  // =========================================================================
447  // Step 6. Set clock dividers for each clock
448  // =========================================================================
449  // Set CPU clock divider
450  sys->CCLKSEL = bit::Insert(sys->CCLKSEL, config.cpu.divider,
451  CpuClockRegister::kDivider);
452  // Set EMC clock divider
453  sys->EMCCLKSEL = bit::Insert(sys->EMCCLKSEL, Value(config.emc_divider),
454  EmcClockRegister::kDivider);
455  // Set Peripheral clock divider
456  sys->PCLKSEL = bit::Insert(sys->PCLKSEL, config.peripheral_divider,
457  PeripheralClockRegister::kDivider);
458  // Set USB clock divider
459  sys->USBCLKSEL = bit::Insert(sys->USBCLKSEL, Value(config.usb.divider),
460  UsbClockRegister::kDivider);
461  // Set SPIFI clock divider
462  sys->SPIFISEL = bit::Insert(sys->SPIFISEL, config.spifi.divider,
463  SpiFiClockRegister::kDivider);
464 
465  cpu_clock_rate_ = cpu / config.cpu.divider;
466  peripheral_clock_rate_ = cpu / config.peripheral_divider;
467  emc_clock_rate_ = cpu / (Value(config.emc_divider) + 1);
468  usb_clock_rate_ = usb / Value(config.usb.divider);
469  spifi_clock_rate_ = spifi / config.spifi.divider;
470  }

472  private:
473  units::frequency::hertz_t SetupPll(volatile uint32_t * control,
474  volatile uint32_t * config,
475  volatile uint32_t * feed,
476  volatile uint32_t * stat,
477  int pll_index) const
478  {
479  const auto & pll_config = clock_configuration_.pll[pll_index];
480  units::frequency::hertz_t fcco;
481 
482  if (pll_config.enabled)
483  {
484  *config = bit::Insert(*config, pll_config.multiply - 1,
485  PllRegister::kMultiplier);
486 
487  if (clock_configuration_.system_oscillator == OscillatorSource::kIrc &&
488  pll_index == 0)
489  {
490  fcco = kDefaultIRCFrequency * pll_config.multiply;
491  }
492  else
493  {
494  fcco = clock_configuration_.external_oscillator_frequency *
495  pll_config.multiply;
496  }
497 
498  // In the datasheet this is the divider, but it acts to multiply the
499  // frequency higher to a point where the fcco is stable.
500  //
501  // fcco must be between 156 MHz to 320 MHz.
502  uint32_t fcco_divide = 0;
503  for (int divide_codes : { 0, 1, 2, 3 })
504  {
505  // Multiply the fcco by 2^divide_code
506  units::frequency::hertz_t final_fcco = fcco * (1 << divide_codes);
507  if (156_MHz <= final_fcco && final_fcco <= 320_MHz)
508  {
509  fcco_divide = divide_codes;
510  break;
511  }
512  }
513 
514  SJ2_ASSERT_FATAL(
515  fcco_divide != 0,
516  "The provided multiply value and oscillator choices results in "
517  "FCCO frequency outside of 156 MHz and 320 MHz. See page 65 "
518  "UM10562 LPC408x/407x User manual for more details.");
519 
520  *config = bit::Insert(*config, fcco_divide, PllRegister::kDivider);
521  // Enable PLL
522  *control = 1;
523  // Feed PLL in order to start the locking process
524  *feed = 0xAA;
525  *feed = 0x55;
526 
527  while (!bit::Read(*stat, PllRegister::kPllLockStatus))
528  {
529  continue;
530  }
531  }
532 
533  return fcco;
534  }

536  void EnableExternalOscillator() const
537  {
538  auto frequency = clock_configuration_.external_oscillator_frequency;
539  if (1_MHz <= frequency && frequency <= 20_MHz)
540  {
541  system_controller->SCS =
542  bit::Clear(system_controller->SCS, OscillatorRegister::kRangeSelect);
543  }
544  else if (20_MHz <= frequency && frequency <= 25_MHz)
545  {
546  system_controller->SCS =
547  bit::Set(system_controller->SCS, OscillatorRegister::kRangeSelect);
548  }
549 
550  SJ2_ASSERT_FATAL(
551  1_MHz <= frequency && frequency <= 25_MHz,
552  "External Oscillator Frequency is outside of the the acceptable 1 "
553  "MHz <--> 25 MHz.");
554 
555  system_controller->SCS =
556  bit::Set(system_controller->SCS, OscillatorRegister::kExternalEnable);
557 
558  while (
559  !bit::Read(system_controller->SCS, OscillatorRegister::kExternalReady))
560  {
561  continue;
562  }
563  }

565  ClockConfiguration & clock_configuration_;
566  units::frequency::hertz_t cpu_clock_rate_ = 0_Hz;
567  units::frequency::hertz_t peripheral_clock_rate_ = 0_Hz;
568  units::frequency::hertz_t emc_clock_rate_ = 0_Hz;
569  units::frequency::hertz_t usb_clock_rate_ = 0_Hz;
570  units::frequency::hertz_t spifi_clock_rate_ = 0_Hz;
