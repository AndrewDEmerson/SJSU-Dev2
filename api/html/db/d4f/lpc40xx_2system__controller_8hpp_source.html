302  uint32_t CalculatePll(units::frequency::megahertz_t input_frequency,
303  units::frequency::megahertz_t desired_frequency) const
304  {
305  SJ2_ASSERT_FATAL(desired_frequency < 384_MHz && desired_frequency > 12_MHz,
306  "Frequency must be lower than 384 MHz"
307  "and greater than or equal to 12 MHz");
308  bool calculating = true;
309  uint32_t multiplier_value;
310  if ((desired_frequency.to<uint32_t>() % input_frequency.to<uint32_t>()) > 0)
311  {
312  multiplier_value = (desired_frequency / input_frequency) + 1;
313  }
314  else
315  {
316  multiplier_value = desired_frequency / input_frequency;
317  }
318 
319  uint16_t divider_value = 1;
320 
321  while (calculating)
322  {
323  uint16_t current_controlled_oscillator_frequency;
324  current_controlled_oscillator_frequency = static_cast<uint16_t>(
325  (static_cast<uint16_t>(input_frequency) * multiplier_value * 2) *
326  divider_value);
327  if (current_controlled_oscillator_frequency >= 156)
328  {
329  calculating = false;
330  }
331  else
332  {
333  divider_value = static_cast<uint16_t>(divider_value * 2);
334  SJ2_ASSERT_FATAL(divider_value < 8,
335  "PLL divider value went out of bounds");
336  }
337  }
338 
339  return multiplier_value;
340  }

342  void SetMainPll(units::frequency::megahertz_t input_frequency,
343  units::frequency::megahertz_t desired_frequency) const
344  {
345  uint16_t divider_value = 1;
346  uint32_t multiplier_value =
347  CalculatePll(input_frequency, desired_frequency);
348  // units::frequency::megahertz_t actual_speed =
349  // input_frequency * multiplier_value;
350  // TO DO: use registers to retreive values
351  SelectOscillatorSource(OscillatorSource::kIrc);
352  SelectMainClockSource(MainClockSource::kBaseClock);
353  SelectUsbClockSource(UsbSource::kBaseClock);
354  SelectSpifiClockSource(SpifiSource::kBaseClock);
355 
356  uint32_t config = 0;
357  // must subtract 1 from multiplier value as specified in datasheet
358  config = bit::Insert(config, multiplier_value - 1, Pll::kMultiplier);
359  config = bit::Insert(config, divider_value, Pll::kDivider);
360  // Set the PLL multiply and divide values
361  system_controller->PLL0CFG = config;
362 
363  // Enable PLL
364  system_controller->PLL0CON =
365  bit::Set(system_controller->PLL0CON, Pll::kEnable);
366 
367  // Necessary feed sequence to ensure the changes are intentional
368  system_controller->PLL0FEED = 0xAA;
369  system_controller->PLL0FEED = 0x55;
370 
371  while (!bit::Read(system_controller->PLL0STAT, Pll::kPllLockStatus))
372  {
373  continue;
374  }
375  }
