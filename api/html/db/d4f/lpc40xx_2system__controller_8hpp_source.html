289  uint32_t CalculatePll(units::frequency::megahertz_t input_frequency,
290  units::frequency::megahertz_t desired_frequency) const
291  {
292  SJ2_ASSERT_FATAL(desired_frequency < 384_MHz && desired_frequency > 12_MHz,
293  "Frequency must be lower than 384 MHz"
294  "and greater than or equal to 12 MHz");
295  bool calculating = true;
296  uint32_t multiplier_value;
297  if ((desired_frequency.to<uint32_t>() % input_frequency.to<uint32_t>()) > 0)
298  {
299  multiplier_value =
300  static_cast<uint32_t>((desired_frequency / input_frequency) + 1);
301  }
302  else
303  {
304  multiplier_value =
305  static_cast<uint32_t>(desired_frequency / input_frequency);
306  }
307  uint16_t divider_value = 1;
308  while (calculating)
309  {
310  uint16_t current_controlled_oscillator_frequency;
311  current_controlled_oscillator_frequency = static_cast<uint16_t>(
312  (static_cast<uint16_t>(input_frequency) * multiplier_value * 2) *
313  divider_value);
314  if (current_controlled_oscillator_frequency >= 156)
315  {
316  calculating = false;
317  }
318  else
319  {
320  divider_value = static_cast<uint16_t>(divider_value * 2);
321  SJ2_ASSERT_FATAL(divider_value < 8,
322  "PLL divider value went out of bounds");
323  }
324  }
325 
326  return multiplier_value;
327  }

329  void SetMainPll(units::frequency::megahertz_t input_frequency,
330  units::frequency::megahertz_t desired_frequency) const
331  {
332  uint16_t divider_value = 1;
333  uint32_t multiplier_value =
334  CalculatePll(input_frequency, desired_frequency);
335  // units::frequency::megahertz_t actual_speed =
336  // input_frequency * multiplier_value;
337  // TO DO: use registers to retreive values
338  SelectOscillatorSource(OscillatorSource::kIrc);
339  SelectMainClockSource(MainClockSource::kBaseClock);
340  SelectUsbClockSource(UsbSource::kBaseClock);
341  SelectSpifiClockSource(SpifiSource::kBaseClock);
342  // must subtract 1 from multiplier value as specified in datasheet
343  system_controller->PLL0CFG =
344  (system_controller->PLL0CFG & ~kClearPllMultiplier) |
345  (multiplier_value - 1);
346  system_controller->PLL0CFG =
347  (system_controller->PLL0CFG & ~kClearPllDivider) | (divider_value << 5);
348  system_controller->PLL0CON |= kEnablePll;
349  // Necessary feed sequence to ensure the changes are intentional
350  system_controller->PLL0FEED = 0xAA;
351  system_controller->PLL0FEED = 0x55;
352 
353  while (!bit::Read(system_controller->PLL0STAT, kPlock))
354  {
355  continue;
356  }
357  }
