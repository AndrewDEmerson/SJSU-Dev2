356  static void CanIrqHandler()
357  {
358  TxMessage_t * message_ptr;
359  bool transmit_interrupt, receive_interrupt;
360 
361  for (uint8_t controller = 0; controller < kNumberOfControllers;
362  controller++)
363  {
364  if (is_controller_initialized[controller] == true)
365  {
366  // Capture interrupt register
367  // Bits are cleared when you read the ICR register
368  interrupts[controller].ICR = can_registers[controller]->ICR;
369 
370  transmit_interrupt = interrupts[controller].flags.tx_buffer_1_ready ||
371  interrupts[controller].flags.tx_buffer_2_ready ||
372  interrupts[controller].flags.tx_buffer_3_ready;
373 
374  receive_interrupt = interrupts[controller].flags.rx_buffer_full;
375 
376  if (transmit_interrupt == true)
377  {
378  if (xQueueIsQueueEmptyFromISR(transmit_queue[controller]) == pdFALSE)
379  {
380  // dequeue message pointer
381  xQueueReceiveFromISR(transmit_queue[controller], &message_ptr, 0);
382  // send it to the available buffer
383  if (interrupts[controller].flags.tx_buffer_1_ready)
384  {
385  WriteMessageToBuffer(
386  message_ptr, kBuffer1, static_cast<Controllers>(controller));
387  }
388  else if (interrupts[controller].flags.tx_buffer_2_ready)
389  {
390  WriteMessageToBuffer(
391  message_ptr, kBuffer2, static_cast<Controllers>(controller));
392  }
393  else if (interrupts[controller].flags.tx_buffer_3_ready)
394  {
395  WriteMessageToBuffer(
396  message_ptr, kBuffer3, static_cast<Controllers>(controller));
397  }
398  else
399  {
400  // will never occur
401  }
402  }
403  }
404 
405  if (receive_interrupt == true)
406  {
407  ReadMessageFromBuffer(static_cast<Controllers>(controller));
408  }
409  }
410  }

482  // TODO(#344):
483  // If this Send() function gets a pointer that only has scope within a
484  // function call, then once that function goes out of scope, the pointer will
485  // become a dangling pointer (pointer to invalid data). Afterwards, when
486  // WriteMessageToBuffer() tries to access the data to write it to the HW
487  // buffers, it will get corrupt data. This problem can arise if the user wants
488  // to send a 1 time message. However, if the user is sending cyclic messages,
489  // it is assumed that global message handles exist somewhere in the user's app
490  // layer since they need to be updated periodically.
491  bool Send(TxMessage_t * const kMessage,
492  uint32_t id,
493  const uint8_t * const kPayload,
494  size_t length) const override
495  {
496  bool success = false;
497 
498  // Capture status flag register
499  status[controller_].SR = can_registers[controller_]->SR;
500 
501  // Assemble the CAN message with the user provided data.
502  //
503  // Since this memory location is written to by this function and is read by
504  // the ISR (to eventually write the data at this memory location into the
505  // transmit buffer), we disable the CAN interrupt to ensure that complete
506  // and valid data is written before it is accessed by the ISR.
507  sjsu::InterruptController::GetPlatformController().Disable(CAN_IRQn);
508  kMessage->id = id;
509  kMessage->frame_data.data_length = (length & 0xF);
510  kMessage->frame_data.remote_tx_request = 0;
511  for (uint8_t i = 0; i < length; i++)
512  {
513  kMessage->data.bytes[i] = kPayload[i];
514  }
515  sjsu::InterruptController::GetPlatformController().Enable(
516  kCanInterruptInfo);
517 
518  // Check if any buffer is available.
519  if (status[controller_].flags.tx_buffer_1_released)
520  {
521  WriteMessageToBuffer(kMessage, kBuffer1, controller_);
522  success = true;
523  }
524  else if (status[controller_].flags.tx_buffer_2_released)
525  {
526  WriteMessageToBuffer(kMessage, kBuffer2, controller_);
527  success = true;
528  }
529  else if (status[controller_].flags.tx_buffer_3_released)
530  {
531  WriteMessageToBuffer(kMessage, kBuffer3, controller_);
532  success = true;
533  }
534  else
535  {
536  // All buffers busy, enqueue the message pointer.
537  // TODO(#345): Ideally this should be a priority queue based on message
538  // ID.
539  if (xQueueSend(transmit_queue[controller_], &kMessage, 0) == pdPASS)
540  {
541  success = true;
542  }
543  }
544  return success;
545  }

557  bool SelfTestBus(uint32_t id) const override
558  {
559  bool success = false;
560 
561  TxMessage_t test_message_tx;
562  RxMessage_t test_message_rx;
563 
564  memset(&test_message_tx, 0, sizeof(TxMessage_t));
565  memset(&test_message_rx, 0, sizeof(RxMessage_t));
566 
567  test_message_tx.id = id;
568 
569  // Enable reset mode in order to write to registers
570  SetControllerMode(kReset, true);
571  // Enable self-test mode
572  SetControllerMode(kSelfTest, true);
573  // Disable reset mode
574  SetControllerMode(kReset, false);
575  // Write test message to tx buffer 1
576  WriteMessageToBuffer(&test_message_tx, kBuffer1, controller_);
577  // Set self-test request and send buffer 1
578  can_registers[controller_]->CMR = kSelfReceptionSendTxBuffer1;
579 
580  // Allow time for RX interrupt to fire
581  sjsu::Delay(2ms);
582 
583  // Get the message; the ISR (interrupt service routine)
584  // will read the message from the rx buffer
585  // and enqueue it into the rx queue.
586  Receive(&test_message_rx);
587 
588  // Check if the received message matches the one we sent
589  if (test_message_rx.id == test_message_tx.id)
590  {
591  success = true;
592  }
593 
594  // Disable self-test mode
595  SetControllerMode(kReset, true);
596  SetControllerMode(kSelfTest, false);
597  SetControllerMode(kReset, false);
598 
599  return success;
600  }

646  private:
647  static void WriteMessageToBuffer(const TxMessage_t * const kMessage,
648  TxBuffers buffer_number,
649  Controllers controller)
650  {
651  volatile TxMessage_t * ptr_to_buffer_offset;
652  Commands command;
653 
654  switch (buffer_number)
655  {
656  case kBuffer1:
657  {
658  ptr_to_buffer_offset = reinterpret_cast<volatile TxMessage_t *>(
659  &can_registers[controller]->TFI1);
660  command = kSendTxBuffer1;
661  }
662  break;
663  case kBuffer2:
664  {
665  ptr_to_buffer_offset = reinterpret_cast<volatile TxMessage_t *>(
666  &can_registers[controller]->TFI2);
667  command = kSendTxBuffer2;
668  }
669  break;
670  case kBuffer3:
671  {
672  ptr_to_buffer_offset = reinterpret_cast<volatile TxMessage_t *>(
673  &can_registers[controller]->TFI3);
674  command = kSendTxBuffer3;
675  }
676  break;
677  default:
678  // should never occur
679  break;
680  }
681 
682  sjsu::InterruptController::GetPlatformController().Disable(CAN_IRQn);
683  ptr_to_buffer_offset->TFI = kMessage->TFI;
684  ptr_to_buffer_offset->id = kMessage->id;
685  ptr_to_buffer_offset->data.qword = kMessage->data.qword;
686  can_registers[controller]->CMR = command;
687  sjsu::InterruptController::GetPlatformController().Enable(
688  kCanInterruptInfo);
689  }

692  {

700  sjsu::InterruptController::GetPlatformController().Disable(CAN_IRQn);

703  message.data.qword = ptr_to_buffer_offset->data.qword;
704  sjsu::InterruptController::GetPlatformController().Enable(
705  kCanInterruptInfo);
706 
707  xQueueSendFromISR(receive_queue[controller], &message, 0);
708  can_registers[controller]->CMR = kReleaseRxBuffer;
709  }

786  if (baud_rate_ == kBaud100Kbps)
787  {
788  // According to the BOSCH CAN spec, the nominal bit time is divided into 4
789  // time segments. These segments need to be programmed for the internal
790  // bit timing logic/state machine. Refer to the link below for more
791  // details about the importance of these time segments:
792  // http://www.keil.com/dd/docs/datashts/silabs/boschcan_ug.pdf
793  //
794  // Nominal Bit Time : 1 / 100 000 == 10^-5s
795  //
796  // 0_______________________________10^-5
797  // _/ 1 bit \_
798  // \_______________________________/
799  //
800  // | SYNC | PROP | PHASE1 | PHASE2 | BOSCH
801  // | TSCL | TSEG1 | TSEG2 | LPC
802  // ^
803  // sample point (industry standard: 80%)
804 
805  // Scaled down to 1Mhz
806  bus_timing.values.baud_rate_prescaler = 47;
807  // Used to compensate for positive and negative edge phase errors. Defines
808  // how much the sample point can be shifted.
809  bus_timing.values.sync_jump_width = 3;
810  // These time segments determine the location of the "sample point".
811  bus_timing.values.time_segment_1 = 6;
812  bus_timing.values.time_segment_2 = 1;
813  // The bus is sampled 3 times (recommended for low speeds).
814  bus_timing.values.sampling = 0;
815 
816  can_registers[controller_]->BTR = bus_timing.BTR;
817  }
