97 constexpr static UartCalibration_t GenerateUartCalibration(
98  uint32_t baud_rate, units::frequency::hertz_t peripheral_frequency)
99 {
100  uint32_t integer_peripheral_frequency =
101  units::unit_cast<uint32_t>(peripheral_frequency);
102  States state = States::kCalculateIntegerDivideLatch;
103  UartCalibration_t uart_calibration;
104  float baud_rate_float = static_cast<float>(baud_rate);
105  float divide_estimate = 0;
106  float decimal = 1.5;
107  float div = 1;
108  float mul = 2;
109  while (state != States::kDone)
110  {
111  switch (state)
112  {
113  case States::kCalculateIntegerDivideLatch:
114  {
115  divide_estimate =
116  DividerEstimate(baud_rate_float, 1, integer_peripheral_frequency);
117 
118  if (divide_estimate < 1.0f)
119  {
120  uart_calibration.divide_latch = 0;
121  state = States::kDone;
122  }
123  else if (IsDecmial(divide_estimate))
124  {
125  uart_calibration.divide_latch =
126  static_cast<uint32_t>(divide_estimate);
127  state = States::kDone;
128  }
129  else
130  {
131  state = States::kCalculateDivideLatchWithDecimal;
132  }
133  break;
134  }
135  case States::kCalculateDivideLatchWithDecimal:
136  {
137  divide_estimate = RoundFloat(DividerEstimate(
138  baud_rate_float, decimal, integer_peripheral_frequency));
139  decimal = FractionalEstimate(
140  baud_rate_float, divide_estimate, integer_peripheral_frequency);
141  if (1.1f <= decimal && decimal <= 1.9f)
142  {
143  state = States::kGenerateFractionFromDecimal;
144  }
145  else
146  {
147  state = States::kDecimalFailedGenerateNewDecimal;
148  }
149  break;
150  }
151  case States::kDecimalFailedGenerateNewDecimal:
152  {
153  mul += 1;
154 
155  if (div > 15)
156  {
157  state = States::kDone;
158  break;
159  }
160  else if (mul > 15)
161  {
162  div += 1;
163  mul = div + 1;
164  }
165  decimal = div / mul;
166  state = States::kCalculateDivideLatchWithDecimal;
167  break;
168  }
169  case States::kGenerateFractionFromDecimal:
170  {
171  uart_calibration = FindClosestFractional(decimal);
172  uart_calibration.divide_latch = static_cast<uint32_t>(divide_estimate);
173  state = States::kDone;
174  break;
175  }
176  case States::kDone:
177  default: break;
178  }
179  }
180  return uart_calibration;
181 }
