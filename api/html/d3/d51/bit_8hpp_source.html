100 template <typename T>
101 [[nodiscard]] constexpr T Extract(T target,
102  uint32_t position,
103  uint32_t width = 1)
104 {
105  // Check the types at compile time
106  static_assert(std::numeric_limits<T>::is_integer,
107  "Extract only accepts intergers.");
108  // Need to use an unsigned version of the type T for the mask to make sure
109  // that the shift right doesn't result in a sign extended shift.
110  using UnsignedT = typename std::make_unsigned<T>::type;
111  // At compile time, generate variable containing all 1s with the size of the
112  // target parameter.
113  constexpr UnsignedT kFieldOfOnes = std::numeric_limits<UnsignedT>::max();
114  // At compile time calculate the number of bits in the target parameter.
115  constexpr size_t kTargetWidth = sizeof(T) * 8;
116  // Create mask by shifting the set of 1s down so that the number of 1s from
117  // bit position 0 is equal to the width parameter.
118  UnsignedT mask =
119  static_cast<UnsignedT>(kFieldOfOnes >> (kTargetWidth - width));
120  // Shift target down to the right to get the bit position in the 0th bit
121  // location.
122  // Mask the value to clear any bit after the width's amount of bits.
123  return static_cast<T>((target >> position) & mask);
124 }

148 template <typename T, typename U>
149 [[nodiscard]] constexpr T Insert(T target,
150  U value,
151  uint32_t position,
152  uint32_t width = 1)
153 {
154  // Check the types at compile time
155  static_assert(
156  std::numeric_limits<T>::is_integer,
157  "1st parameter (target) of Insert function must be an interger.");
158  static_assert(
159  std::numeric_limits<U>::is_integer,
160  "2nd parameter (value) of Insert function must be an interger.");
161  static_assert(sizeof(T) >= sizeof(U),
162  "2nd parameter (value) must be smaller than or equal to the "
163  "size of the target value.");
164  // Need to use an unsigned version of the type T for the mask to make sure
165  // that the shift right doesn't result in a sign extended shift.
166  using UnsignedT = typename std::make_unsigned<T>::type;
167  // At compile time, generate variable containing all 1s with the size of the
168  // target parameter.
169  constexpr UnsignedT kFieldOfOnes = std::numeric_limits<UnsignedT>::max();
170  // At compile time calculate the number of bits in the target parameter.
171  constexpr size_t kTargetWidth = sizeof(T) * 8;
172  // Create mask by shifting the set of 1s down so that the number of 1s from
173  // bit position 0 is equal to the width parameter.
174  UnsignedT mask =
175  static_cast<UnsignedT>(kFieldOfOnes >> (kTargetWidth - width));
176  // Clear width's number of bits in the target value at the bit position
177  // specified.
178  target = static_cast<UnsignedT>(target & ~(mask << position));
179  // AND value with mask to remove any bits beyond the specified width.
180  // Shift masked value into bit position and OR with target value.
181  target = static_cast<UnsignedT>(target | (value & mask) << position);
182  return static_cast<T>(target);
183 }

362 template <typename T>
363 constexpr T StreamExtract(const uint8_t * stream,
364  size_t size,
365  Mask mask,
366  sjsu::Endian endian = sjsu::Endian::kLittle)
367 {
368  static_assert(
369  std::is_same<T, uint8_t>::value || std::is_same<T, uint16_t>::value ||
370  std::is_same<T, uint32_t>::value || std::is_same<T, uint64_t>::value,
371  "only unsigned integers are allowed to be used in "
372  "sjsu::bit::StreamExtract().");
373 
374  // Diving the starting position by 8 will give you the starting byte position.
375  // The fractional part is truncated, which gives the correct answer. If it
376  // were rounded, the answer would be incorrect and would require the use of
377  // the floor() function.
378  const size_t kStartByte = mask.position / 8;
379  // We need to figure out if the bit position is not byte aligned. We can do
380  // this by taking the modulus of the mask's bit starting position.
381  const size_t kStartingBitOffset = mask.position % 8;
382  // We need to figure out if the bit position of the last bits not byte
383  // aligned. This is simply to determine if we need to consume an extra byte
384  // from the stream.
385  const size_t kEndingBitOffset = (mask.position + mask.width) % 8;
386 
387  // Contains the total number of bytes that will need to extracted from the
388  // byte stream. The number of bytes can be calculated by taking the bit width
389  // desired and dividing it by the size of a byte (8-bits).
390  size_t total_bytes = mask.width / 8;
391 
392  if (kStartingBitOffset != 0)
393  {
394  total_bytes++;
395  }
396 
397  if (kEndingBitOffset != 0)
398  {
399  total_bytes++;
400  }
401 
402  T result = 0;
403 
404  // A pointer to the starting byte of the sequence to pull bytes from.
405  const uint8_t * next_byte_ptr = &stream[kStartByte];
406  // The direction in which we will pull bytes. This is assumed BIG Endian,
407  // thus, the LSB is the first byte. Going forward will pull out more
408  // significant bytes.
409  int scan_direction = 1;
410 
411  if (endian == sjsu::Endian::kLittle)
412  {
413  // Little Endian has its LSB at the end of the array, thus we will pull
414  // bytes from the end of the byte array.
415  next_byte_ptr = &stream[(size - 1) - kStartByte];
416  // To get the more significate bytes, we need to progress through the array
417  // backwards, so we need to increase backwards.
418  scan_direction = -1;
419  }
420 
421  for (uint32_t i = 0; i < total_bytes && kStartByte + i < size; i++)
422  {
423  const int kShiftAmount = (8 * i) - kStartingBitOffset;
424  const uint8_t kNextByte = *next_byte_ptr;
425  next_byte_ptr += scan_direction;
426  T finished_byte = 0;
427 
428  if (kShiftAmount > 0)
429  {
430  finished_byte = kNextByte << kShiftAmount;
431  }
432  else
433  {
434  finished_byte = kNextByte >> -kShiftAmount;
435  }
436  result |= finished_byte;
437  }
438 
439  Mask trimming_mask = mask;
440  trimming_mask.position = 0;
441  result = Extract(result, trimming_mask);
442 
443  return result;
444 }
