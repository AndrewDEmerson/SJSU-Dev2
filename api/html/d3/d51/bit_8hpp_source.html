58 [[nodiscard]] constexpr T Extract(T target,
59  uint32_t position,
60  uint32_t width = 1)
61 {
62  // Check the types at compile time
63  static_assert(std::numeric_limits<T>::is_integer,
64  "Extract only accepts intergers.");
65  // Need to use an unsigned version of the type T for the mask to make sure
66  // that the shift right doesn't result in a sign extended shift.
67  using UnsignedT = typename std::make_unsigned<T>::type;
68  // At compile time, generate variable containing all 1s with the size of the
69  // target parameter.
70  constexpr UnsignedT kFieldOfOnes = std::numeric_limits<UnsignedT>::max();
71  // At compile time calculate the number of bits in the target parameter.
72  constexpr size_t kTargetWidth = sizeof(T) * 8;
73  // Create mask by shifting the set of 1s down so that the number of 1s from
74  // bit position 0 is equal to the width parameter.
75  UnsignedT mask = kFieldOfOnes >> (kTargetWidth - width);
76  // Shift target down to the right to get the bit position in the 0th bit
77  // location.
78  // Mask the value to clear any bit after the width's amount of bits.
79  return static_cast<T>((target >> position) & mask);
80 }
