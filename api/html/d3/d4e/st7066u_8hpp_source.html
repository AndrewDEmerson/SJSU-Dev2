170  void WriteNibble(WriteOperation operation, uint8_t nibble)
171  {

177  kControlPins.d6.Set(sjsu::Gpio::State((nibble >> 2) & 0x01));
178  kControlPins.d5.Set(sjsu::Gpio::State((nibble >> 1) & 0x01));
179  kControlPins.d4.Set(sjsu::Gpio::State((nibble >> 0) & 0x01));

219  virtual void WriteCommand(uint8_t command)
220  {
221  switch (kBusMode)
222  {
223  case BusMode::kFourBit:
224  WriteNibble(WriteOperation::kCommand, (command >> 4) & 0xF);
225  WriteNibble(WriteOperation::kCommand, (command >> 0) & 0xF);
226  break;
227  case BusMode::kEightBit:
228  WriteByte(WriteOperation::kCommand, command);
229  break;
230  }

319  virtual void DisplayText(const char * text,
320  CursorPosition_t position = kDefaultCursorPosition)
321  {
322  constexpr uint8_t kMaxDisplayWidth = 20;
323  const uint8_t kStartOffset = position.position;
324  uint8_t termination_index = uint8_t(strlen(text));
325  // calculate the termination_index to stop writing to the display
326  // if the string length of text exceeds the display's width
327  if (termination_index > kMaxDisplayWidth)
328  {
329  termination_index = kMaxDisplayWidth;
330  }
331  if (termination_index > (kMaxDisplayWidth - kStartOffset))
332  {
333  termination_index = uint8_t(kMaxDisplayWidth - kStartOffset);
334  }
335  // set cursor start position for writing
336  SetCursorPosition(position);
337  for (uint8_t i = 0; i < termination_index; i++)
338  {
339  WriteData(text[i]);
340  }
341  }
