215  void SetSystemClockFrequency(units::frequency::megahertz_t frequency) const
216  {
217  // The following sequence is specified in the LPC176x/5x User Manual
218  // Section 4.5.13 and must be followed step by step in order to have PLL0
219  // initialized and running.
220 
221  // NOTE: It is very important not to merge any steps. For example, do not
222  // update the PLL0CFG and enable PLL0 simultaneously with the same
223  // feed sequence.
224 
225  // 1. Disconnect PLL0 with one feed sequence if PLL0 is already connected.
226  system_controller->PLL0CON =
227  bit::Clear(system_controller->PLL0CON, Pll::kConnectBit);
228  WritePllFeedSequence(PllSelect::kMainPll);
229  // 2. Disable PLL0 with one feed sequence.
230  system_controller->PLL0CON =
231  bit::Clear(system_controller->PLL0CON, Pll::kEnableBit);
232  WritePllFeedSequence(PllSelect::kMainPll);
233  // 3. Change the CPU Clock Divider setting to speed up operation without
234  // PLL0, if desired.
235  SetCpuClockDivider(0);
236  // 4. Write to the Clock Source Selection Control register to change the
237  // clock source if needed.
238  // The 4 MHz internal RC will be used until PLL0 achieves a lock and is
239  // connected.
240  SelectOscillatorSource(OscillatorSource::kIrc);
241  // 5. Write to the PLL0CFG and make it effective with one feed sequence.
242  // The PLL0CFG can only be updated when PLL0 is disabled.
243  const PllSettings_t kPll0Settings =
244  CalculatePll0(kDefaultIRCFrequency, frequency);
245  system_controller->PLL0CFG =
246  bit::Insert(system_controller->PLL0CFG, kPll0Settings.multiplier,
247  MainPll::kMultiplier);
248  system_controller->PLL0CFG =
249  bit::Insert(system_controller->PLL0CFG, kPll0Settings.pre_divider,
250  MainPll::kPreDivider);
251  WritePllFeedSequence(PllSelect::kMainPll);
252  // 6. Enable PLL0 with one feed sequence.
253  system_controller->PLL0CON =
254  bit::Set(system_controller->PLL0CON, Pll::kEnableBit);
255  WritePllFeedSequence(PllSelect::kMainPll);
256  // 7. Change the CPU Clock Divider setting for the operation with PLL0.
257  // It is critical to do this before connecting PLL0.
258  SetCpuClockDivider(kPll0Settings.cpu_divider);
259  // 8. Wait for PLL0 to achieve lock by monitoring the PLOCK0 bit in the
260  // PLL0STAT register, or using the PLOCK0 interrupt, or wait for a fixed
261  // time when the input clock to PLL0 is slow (i.e. 32 kHz).
262  SJ2_ASSERT_FATAL(WaitForPllLockStatus(PllSelect::kMainPll),
263  "PLL0 lock could not be established before timeout");
264  // 9. Connect PLL0 with one feed sequence.
265  system_controller->PLL0CON =
266  bit::Set(system_controller->PLL0CON, Pll::kConnectBit);
267  WritePllFeedSequence(PllSelect::kMainPll);
268 
269  SJ2_ASSERT_FATAL(WaitForPllConnectionStatus(PllSelect::kMainPll),
270  "Failed to connect PLL0.");
271 
272  speed_in_hertz = frequency;
273  }
278  void SetUsbPllInputFrequency(UsbPllInputFrequency frequency)
279  {
280  // NOTE: It is very important not to merge any steps below. For example, do
281  // not update the PLL1CFG and enable PLL1 simultaneously with the
282  // same feed sequence.
283 
284  // 1. Disconnect PLL1 with one feed sequence if PLL1 is already connected.
285  system_controller->PLL1CON =
286  bit::Clear(system_controller->PLL1CON, Pll::kConnectBit);
287  WritePllFeedSequence(PllSelect::kUsbPll);
288  // 2. Disable PLL1 with one feed sequence.
289  system_controller->PLL1CON =
290  bit::Clear(system_controller->PLL1CON, Pll::kEnableBit);
291  WritePllFeedSequence(PllSelect::kUsbPll);
292  // 3. Write to the PLL1CFG and make it effective with one feed sequence.
293  constexpr uint8_t kDivider = Value(UsbPllDivider::kDivideBy1);
294  // depending on the input frequency, the multiplier value will always be
295  // either 2, 3, or 4
296  const uint8_t kMultiplier[] = {
297  Value(UsbPllMultiplier::kMultiplyBy4),
298  Value(UsbPllMultiplier::kMultiplyBy3),
299  Value(UsbPllMultiplier::kMultiplyBy2),
300  };
301  system_controller->PLL1CFG =
302  bit::Insert(system_controller->PLL1CFG, kMultiplier[Value(frequency)],
303  UsbPll::kMultiplier);
304  system_controller->PLL1CFG =
305  bit::Insert(system_controller->PLL1CFG, kDivider, UsbPll::kDivider);
306  WritePllFeedSequence(PllSelect::kUsbPll);
307  // 4. Enable PLL1 with one feed sequence.
308  system_controller->PLL1CON =
309  bit::Set(system_controller->PLL1CON, Pll::kEnableBit);
310  WritePllFeedSequence(PllSelect::kUsbPll);
311  // 5. Configurations to the PLL must be locked before it can be connected.
312  SJ2_ASSERT_FATAL(WaitForPllLockStatus(PllSelect::kUsbPll),
313  "PLL1 lock could not be established before timeout");
314  // 6. Connect PLL1 with one feed sequence.
315  system_controller->PLL1CON =
316  bit::Set(system_controller->PLL1CON, Pll::kConnectBit);
317  WritePllFeedSequence(PllSelect::kUsbPll);
318 
319  SJ2_ASSERT_FATAL(WaitForPllConnectionStatus(PllSelect::kUsbPll),
320  "Failed to connect PLL1.");
321  }
330  void SetPeripheralClockDivider(PeripheralID peripheral_select,
331  uint8_t peripheral_divider) const
332  {
333  const bool kIsCanPeripheral =
334  peripheral_select.device_id == Peripherals::kCan1.device_id ||
335  peripheral_select.device_id == Peripherals::kCan2.device_id;
336  // Convert the divider value to corresponding 2-bit select value
337  // The list of divider select values can be found in the LPC176x/5x User
338  // Manual Table 42.
339  uint8_t divider_select;
340  switch (peripheral_divider)
341  {
342  case 1: divider_select = 0b01; break;
343  case 2: divider_select = 0b10; break;
344  case 4: divider_select = 0b00; break;
345  case 6:
346  {
347  SJ2_ASSERT_FATAL(
348  kIsCanPeripheral,
349  "The divider value of 6 is only supported for CAN peripherals.");
350  divider_select = 0b11;
351  }
352  break;
353  case 8:
354  {
355  SJ2_ASSERT_FATAL(
356  !kIsCanPeripheral,
357  "The divider value of 8 is not supported for CAN peripherals.");
358  divider_select = 0b11;
359  }
360  break;
361  default:
362  SJ2_ASSERT_FATAL(
363  false,
364  "Only the following peripheral divider values are supported: 1, 2, "
365  "4, 8. The divider value of 6 is supported for CAN peripherals.");
366  divider_select = -1;
367  }
368  volatile uint32_t * pclk_sel =
369  GetPeripheralClockSelectRegister(peripheral_select);
370  const bit::Mask kDividerMask =
371  CalculatePeripheralClockDividerMask(peripheral_select);
372  *pclk_sel = bit::Insert(*pclk_sel, divider_select, kDividerMask);
373  }

376  uint32_t GetPeripheralClockDivider(
377  PeripheralID peripheral_select) const
378  {
379  volatile uint32_t * pclk_sel =
380  GetPeripheralClockSelectRegister(peripheral_select);
381  const bit::Mask kDividerMask =
382  CalculatePeripheralClockDividerMask(peripheral_select);
383  const uint8_t kDividerSelect =
384  static_cast<uint8_t>(bit::Extract(*pclk_sel, kDividerMask));
385 
386  uint8_t peripheral_clock_divider;
387  // convert and return the actual peripheral divider value based on the 2-bit
388  // divider select value
389  switch (kDividerSelect)
390  {
391  case 0b00: peripheral_clock_divider = 4; break;
392  case 0b01: peripheral_clock_divider = 1; break;
393  case 0b10: peripheral_clock_divider = 2; break;
394  case 0b11:
395  {
396  // 0b11 for CAN peripiherals use a divider of 6 while all others use a
397  // divider of 8
398  switch (peripheral_select.device_id)
399  {
400  case Peripherals::kCan1.device_id: [[fallthrough]];
401  case Peripherals::kCan2.device_id:
402  peripheral_clock_divider = 6;
403  break;
404  default: peripheral_clock_divider = 8; break;
405  }
406  }
407  }
408  return peripheral_clock_divider;
409  }

454  PllSettings_t CalculatePll0(units::frequency::hertz_t input_frequency,
455  units::frequency::hertz_t desired_speed) const
456  {
457  // minimum/maximum input and output frequencies of PLL0 in kHz
458  constexpr units::frequency::hertz_t kMinimumPll0InputFrequency = 32_kHz;
459  constexpr units::frequency::hertz_t kMaximumPll0InputFrequency = 50_MHz;
460  constexpr units::frequency::hertz_t kMinimumPll0OuputFrequency = 275_MHz;
461  constexpr units::frequency::hertz_t kMaximumPll0OuputFrequency = 550_MHz;
462 
463  // Maximum allowed CPU speed in kHz.
464  // This value will be 100 MHz or 120 MHz depending on the MCU in use
465  // For the SJOne, the max CPU speed for LPC1758 is 100 MHz.
466  constexpr units::frequency::hertz_t kMaxCPUSpeed = 100_MHz;
467 
468  SJ2_ASSERT_FATAL(
469  input_frequency > kMinimumPll0InputFrequency &&
470  input_frequency < kMaximumPll0InputFrequency,
471  "The input PLL0 frequency must be between 32kHz and 50MHz");
472  SJ2_ASSERT_FATAL(
473  desired_speed < kMaxCPUSpeed,
474  "The desired CPU speed cannot exceed the maximum allow CPU speed.");
475  // The supported pre-divider values ranges from 1 to 32 while the supported
476  // multiplier values ranges from 6 to 512.
477  // Since a small value for the pre-divider, n, is desired, we will iterate
478  // through n starting from the lowest possible value of 1 in order to find a
479  // suitable multiplier, m. The values of m and n are inversely proportional;
480  // therefore, we start looking for the multiplier from its largest possible
481  // value of 512.
482  for (uint8_t n = 0; n < 32; n++)
483  {
484  for (uint16_t m = 511; m >= 6; m--)
485  {
486  // Current calculated controlled oscillator frequency, fcco, output of
487  // PLL0 in kilohertz
488  // Dividing by 1000 to scale down kFcco, as the multiplier for is
489  // internally scaled by 1000.
490  const units::frequency::hertz_t kFcco =
491  (2 * (m + 1) * input_frequency) / (n + 1);
492  if (kMinimumPll0OuputFrequency < kFcco &&
493  kFcco < kMaximumPll0OuputFrequency)
494  {
495  // since PLL0 is in use, the cpu_divider values of 0 and 1 are not
496  // allowed as the resulting CPU clock will always be above the maximum
497  // allowed CPU speed
498  for (uint16_t cpu_divider = 2; cpu_divider < 256; cpu_divider++)
499  {
500  // Get resulting CPU clock
501  // Requires that we scale kFcco back to a proper frequency.
502  const units::frequency::hertz_t kCpuClock =
503  kFcco / (cpu_divider + 1);
504  if (kCpuClock == desired_speed)
505  {
506  return PllSettings_t{ .multiplier = m,
507  .pre_divider = n,
508  .cpu_divider =
509  static_cast<uint8_t>(cpu_divider) };
510  }
511  } // cpu_divider loop
512  }
513  } // m for loop
514  } // n for loop
515  SJ2_ASSERT_FATAL(
516  false,
517  "Failed to calculate the PLL0 settings for the desired frequency.");
518  return PllSettings_t{ .multiplier = 0, .pre_divider = 0, .cpu_divider = 0 };
519  }
