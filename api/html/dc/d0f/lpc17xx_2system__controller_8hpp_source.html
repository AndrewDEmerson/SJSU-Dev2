366  void Initialize() override
367  {
368  units::frequency::hertz_t osc_clk =
369  clock_configuration_.main_oscillator.frequency;
370  units::frequency::hertz_t sys_clk = 0_MHz;
371  units::frequency::hertz_t pll_clk = 0_MHz;
372  units::frequency::hertz_t usb_clk = 0_MHz;
373  uint8_t cpu_clock_divider = clock_configuration_.cpu.divider;
374 
375  // =========================================================================
376  // Step 1. Set the internal RC as the clock source for all clocks.
377  // =========================================================================
378  // Disabling and disconnecting the PLLs sets the CPU and USB clock source to
379  // be the system clock.
380  DisableAndDisconnectPll(&system_controller->PLL0CON,
381  &system_controller->PLL0FEED);
382  DisableAndDisconnectPll(&system_controller->PLL1CON,
383  &system_controller->PLL1FEED);
384  // Disable the main oscillator.
385  system_controller->SCS = bit::Clear(
386  system_controller->SCS, SystemControlsRegister::kOscillatorEnableMask);
387  SetSystemClockSource(Oscillator::kIrc);
388 
389  // =========================================================================
390  // Step 2. Enable the main oscillator if it is used to drive the system
391  // clock and/or PLL1.
392  // =========================================================================
393  if ((clock_configuration_.usb.clock_source == UsbClockSource::kPll1) ||
394  (clock_configuration_.system.clock_source == Oscillator::kMain))
395  {
396  EnableMainOscillator();
397  }
398 
399  // =========================================================================
400  // Step 3. Determine the running system clock rate based on the selected
401  // oscillator.
402  // =========================================================================
403  switch (clock_configuration_.system.clock_source)
404  {
405  case Oscillator::kIrc: sys_clk = kIrcFrequency; break;
406  case Oscillator::kMain: sys_clk = osc_clk; break;
407  case Oscillator::kRtc: sys_clk = kRTCFrequency; break;
408  }
409 
410  // =========================================================================
411  // Step 4. Configure the CPU clock by configuring and enabling PLL0 if
412  // necessary.
413  // =========================================================================
414  switch (clock_configuration_.cpu.clock_source)
415  {
416  case SystemController::CpuClockSource::kSystemClock:
417  {
418  SetSystemClockSource(clock_configuration_.system.clock_source);
419  SetCpuClockDivider(cpu_clock_divider);
420  pll_clk = sys_clk;
421  break;
422  }
423  case SystemController::CpuClockSource::kPll0:
424  {
425  // Before configuring PLL0, verify the PLL0 configurations by checking
426  // whether the expected PLL0 current controlled oscillator frequency and
427  // the expected CPU frequency are within the valid frequency ranges.
428 
429  // Minimum / maximum current controlled oscillator frequencies of PLL0.
430  constexpr units::frequency::hertz_t kMinFcco = 275_MHz;
431  constexpr units::frequency::hertz_t kMaxFcco = 550_MHz;
432 
433  uint16_t multiplier = clock_configuration_.pll0.multiplier;
434  uint8_t pre_divider = clock_configuration_.pll0.pre_divider;
435  // Determine the running current controlled oscillator frequency based
436  // on the configurations.
437  pll_clk = (2 * sys_clk * multiplier) / pre_divider;
438 
439  SJ2_ASSERT_FATAL(
440  (kMinFcco < pll_clk) && (pll_clk < kMaxFcco),
441  "Invalid PLL0 multiplier and pre-divider configurations. The "
442  "target current controlled oscillator frequency must be between "
443  "275 MHz and 550 MHz.");
444 
445  auto max_cpu_clock_rate = clock_configuration_.cpu.max_cpu_clock_rate;
446 
447  SJ2_ASSERT_FATAL((pll_clk / cpu_clock_divider) < max_cpu_clock_rate,
448  "The configured CPU divider produces a CPU clock that "
449  "exceeds the maximum allowed CPU clock rate.");
450 
451  ConfigurePll0();
452  break;
453  }
454  }
455 
456  // =========================================================================
457  // Step 5. Configure the USB clock by configuring and enabling PLL1 if
458  // necessary.
459  // =========================================================================
460  switch (clock_configuration_.usb.clock_source)
461  {
462  case UsbClockSource::kPll0:
463  {
464  SJ2_ASSERT_FATAL(
465  clock_configuration_.cpu.clock_source == CpuClockSource::kPll0,
466  "The CPU clock source must be PLL0 in order to use PLL0 as a clock "
467  "source for the USB clock.");
468 
469  uint8_t usb_clock_divider_select =
470  Value(clock_configuration_.usb.divider);
471  uint32_t usb_clock_divider = usb_clock_divider_select + 1;
472  usb_clk = pll_clk / usb_clock_divider;
473 
474  SJ2_ASSERT_FATAL(usb_clk == kUsbClockFrequency,
475  "Attempting to use PLL0 to drive the USB clock, but "
476  "the PLL0 configuration and USB clock divider are not "
477  "configured to produce a frequency of 48 MHz.");
478 
479  system_controller->USBCLKCFG =
480  bit::Insert(system_controller->USBCLKCFG, usb_clock_divider_select,
481  UsbClockRegister::kDividerMask);
482  break;
483  }
484  case UsbClockSource::kPll1:
485  {
486  uint32_t multiplier = Value(clock_configuration_.pll1.multiplier) + 1;
487  usb_clk = osc_clk * multiplier;
488 
489  ConfigurePll1();
490  break;
491  }
492  }
493 
494  // =========================================================================
495  // Step 6. Set all peripheral clock dividers to 1.
496  // =========================================================================
497  // SEE: 4.7.3 Peripheral Clock Selection registers 0 and 1
498  // https://www.nxp.com/docs/en/user-guide/UM10360.pdf#page=58
499  //
500  // NOTE: The reserved bits are left as 0b00.
501  system_controller->PCLKSEL0 = 0x5551'5155;
502  system_controller->PCLKSEL0 = 0x5455'5455;
503 
504  // =========================================================================
505  // Step 7. Determine the running CPU and USB clock rates.
506  // =========================================================================
507  cpu_clock_rate_ = pll_clk / cpu_clock_divider;
508  usb_clock_rate_ = usb_clk;
509  }

571 
572  constexpr units::frequency::hertz_t kMinFrequency = 1_MHz;
573  constexpr units::frequency::hertz_t kMaxFrequency = 25_MHz;
574  SJ2_ASSERT_FATAL(
575  (kMinFrequency <= frequency) && (frequency <= kMaxFrequency),
576  "The frequency of the main oscillator must be between 1 "

580  {
581  system_controls = bit::Clear(

585  {
586  system_controls = bit::Set(system_controls,
587  SystemControlsRegister::kOscillatorRangeMask);

590  SystemControlsRegister::kOscillatorEnableMask);
591 
592  system_controller->SCS = system_controls;
593 
594  // Wait for the main oscillator to become stable before proceeding with any
595  // other operations.
596  while (!bit::Read(system_controller->SCS,
597  SystemControlsRegister::kOscillatorStatusMask))
598  {
599  continue;
600  }
601  }

673  void ConfigurePll0() const
674  {
675  volatile uint32_t * pll_feed_register = &system_controller->PLL0FEED;
676  volatile uint32_t * pll_status_register = &system_controller->PLL0STAT;
677 
678  // The following sequence is specified in the LPC176x/5x User Manual
679  // Section 4.5.13 and must be followed step by step in order to have PLL0
680  // initialized and running.
681 
682  // NOTE: It is very important not to merge any steps. For example, do not
683  // update the PLL0CFG and enable PLL0 simultaneously with the same
684  // feed sequence.
685 
686  // =========================================================================
687  // Step 1. Disconnect PLL0 with one feed sequence if PLL0 is already
688  // connected.
689  //
690  // Step 2. Disable PLL0 with one feed sequence.
691  // =========================================================================
692  // NOTE: DisableAndDisconnectPll performs Steps 1 and 2.
693  DisableAndDisconnectPll(&system_controller->PLL0CON, pll_feed_register);
694 
695  // =========================================================================
696  // Step 3. Change the CPU Clock Divider setting to speed up operation
697  // without PLL0, if desired.
698  // =========================================================================
699  SetCpuClockDivider(1);
700 
701  // =========================================================================
702  // Step 4. Write to the Clock Source Selection Control register to change
703  // the clock source if needed.
704  // =========================================================================
705  SetSystemClockSource(clock_configuration_.system.clock_source);
706 
707  // =========================================================================
708  // Step 5. Write to the PLL0CFG and make it effective with one feed
709  // sequence. The PLL0CFG can only be updated when PLL0 is disabled.
710  // =========================================================================
711  uint16_t multiplier = clock_configuration_.pll0.multiplier;
712  uint8_t pre_divider = clock_configuration_.pll0.pre_divider;
713  system_controller->PLL0CFG =
714  bit::Insert(system_controller->PLL0CFG, (multiplier - 1),
715  Pll0::ConfigurationRegister::kMultiplierMask);
716  system_controller->PLL0CFG =
717  bit::Insert(system_controller->PLL0CFG, (pre_divider - 1),
718  Pll0::ConfigurationRegister::kPreDividerMask);
719  WritePllFeedSequence(pll_feed_register);
720 
721  // =========================================================================
722  // Step 6. Enable PLL0 with one feed sequence.
723  // =========================================================================
724  system_controller->PLL0CON =
725  bit::Set(system_controller->PLL0CON, PllControlRegister::kEnableMask);
726  WritePllFeedSequence(pll_feed_register);
727 
728  // =========================================================================
729  // Step 7. Change the CPU Clock Divider setting for the operation with PLL0.
730  // It is critical to do this before connecting PLL0.
731  // =========================================================================
732  SetCpuClockDivider(clock_configuration_.cpu.divider);
733 
734  // =========================================================================
735  // Step 8. Wait for PLL0 to achieve lock by monitoring the PLOCK0 bit in the
736  // PLL0STAT register, or using the PLOCK0 interrupt, or wait for a
737  // fixed time when the input clock to PLL0 is slow (i.e. 32 kHz).
738  // =========================================================================
739  WaitForPllLockStatus(pll_status_register,
740  Pll0::StatusRegister::kLockStatusMask);
741 
742  // =========================================================================
743  // Step 9. Connect PLL0 with one feed sequence.
744  // =========================================================================
745  system_controller->PLL0CON =
746  bit::Set(system_controller->PLL0CON, PllControlRegister::kConnectMask);
747  WritePllFeedSequence(pll_feed_register);
748 
749  WaitForPllConnectionStatus(pll_status_register,
750  Pll0::StatusRegister::kEnableMask,
751  Pll0::StatusRegister::kConnectMask);
752  }
