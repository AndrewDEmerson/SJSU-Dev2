195  void SetSystemClockFrequency(
196  units::frequency::megahertz_t frequency) const override
197  {
198  // The following sequence is specified in the LPC176x/5x User Manual
199  // Section 4.5.13 and must be followed step by step in order to have PLL0
200  // initialized and running.
201 
202  // NOTE: It is very important not to merge any steps. For example, do not
203  // update the PLL0CFG and enable PLL0 simultaneously with the same
204  // feed sequence.
205 
206  // 1. Disconnect PLL0 with one feed sequence if PLL0 is already connected.
207  system_controller->PLL0CON =
208  bit::Clear(system_controller->PLL0CON, Pll::kConnectBit);
209  WritePllFeedSequence(PllSelect::kMainPll);
210  // 2. Disable PLL0 with one feed sequence.
211  system_controller->PLL0CON =
212  bit::Clear(system_controller->PLL0CON, Pll::kEnableBit);
213  WritePllFeedSequence(PllSelect::kMainPll);
214  // 3. Change the CPU Clock Divider setting to speed up operation without
215  // PLL0, if desired.
216  SetCpuClockDivider(0);
217  // 4. Write to the Clock Source Selection Control register to change the
218  // clock source if needed.
219  // The 4 MHz internal RC will be used until PLL0 achieves a lock and is
220  // connected.
221  SelectOscillatorSource(OscillatorSource::kIrc);
222  // 5. Write to the PLL0CFG and make it effective with one feed sequence.
223  // The PLL0CFG can only be updated when PLL0 is disabled.
224  const PllSettings_t kPll0Settings =
225  CalculatePll0(kDefaultIRCFrequency, frequency);
226  system_controller->PLL0CFG = bit::Insert(system_controller->PLL0CFG,
227  kPll0Settings.multiplier,
228  MainPll::kMultiplier);
229  system_controller->PLL0CFG = bit::Insert(system_controller->PLL0CFG,
230  kPll0Settings.pre_divider,
231  MainPll::kPreDivider);
232  WritePllFeedSequence(PllSelect::kMainPll);
233  // 6. Enable PLL0 with one feed sequence.
234  system_controller->PLL0CON =
235  bit::Set(system_controller->PLL0CON, Pll::kEnableBit);
236  WritePllFeedSequence(PllSelect::kMainPll);
237  // 7. Change the CPU Clock Divider setting for the operation with PLL0.
238  // It is critical to do this before connecting PLL0.
239  SetCpuClockDivider(kPll0Settings.cpu_divider);
240  // 8. Wait for PLL0 to achieve lock by monitoring the PLOCK0 bit in the
241  // PLL0STAT register, or using the PLOCK0 interrupt, or wait for a fixed
242  // time when the input clock to PLL0 is slow (i.e. 32 kHz).
243  SJ2_ASSERT_FATAL(WaitForPllLockStatus(PllSelect::kMainPll),
244  "PLL0 lock could not be established before timeout");
245  // 9. Connect PLL0 with one feed sequence.
246  system_controller->PLL0CON =
247  bit::Set(system_controller->PLL0CON, Pll::kConnectBit);
248  WritePllFeedSequence(PllSelect::kMainPll);
249 
250  SJ2_ASSERT_FATAL(WaitForPllConnectionStatus(PllSelect::kMainPll),
251  "Failed to connect PLL0.");
252 
253  speed_in_hertz = frequency;
254  }
259  void SetUsbPllInputFrequency(UsbPllInputFrequency frequency)
260  {
261  // NOTE: It is very important not to merge any steps below. For example, do
262  // not update the PLL1CFG and enable PLL1 simultaneously with the
263  // same feed sequence.
264 
265  // 1. Disconnect PLL1 with one feed sequence if PLL1 is already connected.
266  system_controller->PLL1CON =
267  bit::Clear(system_controller->PLL1CON, Pll::kConnectBit);
268  WritePllFeedSequence(PllSelect::kUsbPll);
269  // 2. Disable PLL1 with one feed sequence.
270  system_controller->PLL1CON =
271  bit::Clear(system_controller->PLL1CON, Pll::kEnableBit);
272  WritePllFeedSequence(PllSelect::kUsbPll);
273  // 3. Write to the PLL1CFG and make it effective with one feed sequence.
274  constexpr uint8_t kDivider = Value(UsbPllDivider::kDivideBy1);
275  // depending on the input frequency, the multiplier value will always be
276  // either 2, 3, or 4
277  const uint8_t kMultiplier[] = {
278  Value(UsbPllMultiplier::kMultiplyBy4),
279  Value(UsbPllMultiplier::kMultiplyBy3),
280  Value(UsbPllMultiplier::kMultiplyBy2),
281  };
282  system_controller->PLL1CFG = bit::Insert(system_controller->PLL1CFG,
283  kMultiplier[Value(frequency)],
284  UsbPll::kMultiplier);
285  system_controller->PLL1CFG =
286  bit::Insert(system_controller->PLL1CFG, kDivider, UsbPll::kDivider);
287  WritePllFeedSequence(PllSelect::kUsbPll);
288  // 4. Enable PLL1 with one feed sequence.
289  system_controller->PLL1CON =
290  bit::Set(system_controller->PLL1CON, Pll::kEnableBit);
291  WritePllFeedSequence(PllSelect::kUsbPll);
292  // 5. Configurations to the PLL must be locked before it can be connected.
293  SJ2_ASSERT_FATAL(WaitForPllLockStatus(PllSelect::kUsbPll),
294  "PLL1 lock could not be established before timeout");
295  // 6. Connect PLL1 with one feed sequence.
296  system_controller->PLL1CON =
297  bit::Set(system_controller->PLL1CON, Pll::kConnectBit);
298  WritePllFeedSequence(PllSelect::kUsbPll);
299 
300  SJ2_ASSERT_FATAL(WaitForPllConnectionStatus(PllSelect::kUsbPll),
301  "Failed to connect PLL1.");
302  }
311  void SetPeripheralClockDivider(const PeripheralID & peripheral_select,
312  uint8_t peripheral_divider) const override
313  {
314  const bool kIsCanPeripheral =
315  peripheral_select.device_id == Peripherals::kCan1.device_id ||
316  peripheral_select.device_id == Peripherals::kCan2.device_id;
317  // Convert the divider value to corresponding 2-bit select value
318  // The list of divider select values can be found in the LPC176x/5x User
319  // Manual Table 42.
320  uint8_t divider_select;
321  switch (peripheral_divider)
322  {
323  case 1: divider_select = 0b01; break;
324  case 2: divider_select = 0b10; break;
325  case 4: divider_select = 0b00; break;
326  case 6:
327  {
328  SJ2_ASSERT_FATAL(
329  kIsCanPeripheral,
330  "The divider value of 6 is only supported for CAN peripherals.");
331  divider_select = 0b11;
332  }
333  break;
334  case 8:
335  {
336  SJ2_ASSERT_FATAL(
337  !kIsCanPeripheral,
338  "The divider value of 8 is not supported for CAN peripherals.");
339  divider_select = 0b11;
340  }
341  break;
342  default:
343  SJ2_ASSERT_FATAL(
344  false,
345  "Only the following peripheral divider values are supported: 1, 2, "
346  "4, 8. The divider value of 6 is supported for CAN peripherals.");
347  divider_select = -1;
348  }
349  volatile uint32_t * pclk_sel =
350  GetPeripheralClockSelectRegister(peripheral_select);
351  const bit::Mask kDividerMask =
352  CalculatePeripheralClockDividerMask(peripheral_select);
353  *pclk_sel = bit::Insert(*pclk_sel, divider_select, kDividerMask);
354  }

357  uint32_t GetPeripheralClockDivider(
358  const PeripheralID & peripheral_select) const override
359  {
360  volatile uint32_t * pclk_sel =
361  GetPeripheralClockSelectRegister(peripheral_select);
362  const bit::Mask kDividerMask =
363  CalculatePeripheralClockDividerMask(peripheral_select);
364  const uint8_t kDividerSelect =
365  static_cast<uint8_t>(bit::Extract(*pclk_sel, kDividerMask));
366 
367  uint8_t peripheral_clock_divider;
368  // convert and return the actual peripheral divider value based on the 2-bit
369  // divider select value
370  switch (kDividerSelect)
371  {
372  case 0b00: peripheral_clock_divider = 4; break;
373  case 0b01: peripheral_clock_divider = 1; break;
374  case 0b10: peripheral_clock_divider = 2; break;
375  case 0b11:
376  {
377  // 0b11 for CAN peripiherals use a divider of 6 while all others use a
378  // divider of 8
379  switch (peripheral_select.device_id)
380  {
381  case Peripherals::kCan1.device_id: [[fallthrough]];
382  case Peripherals::kCan2.device_id:
383  peripheral_clock_divider = 6;
384  break;
385  default: peripheral_clock_divider = 8; break;
386  }
387  }
388  }
389  return peripheral_clock_divider;
390  }

434  PllSettings_t CalculatePll0(units::frequency::hertz_t input_frequency,
435  units::frequency::hertz_t desired_speed) const
436  {
437  // minimum/maximum input and output frequencies of PLL0 in kHz
438  constexpr units::frequency::hertz_t kMinimumPll0InputFrequency = 32_kHz;
439  constexpr units::frequency::hertz_t kMaximumPll0InputFrequency = 50_MHz;
440  constexpr units::frequency::hertz_t kMinimumPll0OuputFrequency = 275_MHz;
441  constexpr units::frequency::hertz_t kMaximumPll0OuputFrequency = 550_MHz;
442 
443  // Maximum allowed CPU speed in kHz.
444  // This value will be 100 MHz or 120 MHz depending on the MCU in use
445  // For the SJOne, the max CPU speed for LPC1758 is 100 MHz.
446  constexpr units::frequency::hertz_t kMaxCPUSpeed = 100_MHz;
447 
448  SJ2_ASSERT_FATAL(
449  input_frequency > kMinimumPll0InputFrequency &&
450  input_frequency < kMaximumPll0InputFrequency,
451  "The input PLL0 frequency must be between 32kHz and 50MHz");
452  SJ2_ASSERT_FATAL(
453  desired_speed < kMaxCPUSpeed,
454  "The desired CPU speed cannot exceed the maximum allow CPU speed.");
455  // The supported pre-divider values ranges from 1 to 32 while the supported
456  // multiplier values ranges from 6 to 512.
457  // Since a small value for the pre-divider, n, is desired, we will iterate
458  // through n starting from the lowest possible value of 1 in order to find a
459  // suitable multiplier, m. The values of m and n are inversely proportional;
460  // therefore, we start looking for the multiplier from its largest possible
461  // value of 512.
462  for (uint8_t n = 0; n < 32; n++)
463  {
464  for (uint16_t m = 511; m >= 6; m--)
465  {
466  // Current calculated controlled oscillator frequency, fcco, output of
467  // PLL0 in kilohertz
468  // Dividing by 1000 to scale down kFcco, as the multiplier for is
469  // internally scaled by 1000.
470  const units::frequency::hertz_t kFcco =
471  (2 * (m + 1) * input_frequency) / (n + 1);
472  if (kMinimumPll0OuputFrequency < kFcco &&
473  kFcco < kMaximumPll0OuputFrequency)
474  {
475  // since PLL0 is in use, the cpu_divider values of 0 and 1 are not
476  // allowed as the resulting CPU clock will always be above the maximum
477  // allowed CPU speed
478  for (uint16_t cpu_divider = 2; cpu_divider < 256; cpu_divider++)
479  {
480  // Get resulting CPU clock
481  // Requires that we scale kFcco back to a proper frequency.
482  const units::frequency::hertz_t kCpuClock =
483  kFcco / (cpu_divider + 1);
484  if (kCpuClock == desired_speed)
485  {
486  return PllSettings_t{ .multiplier = m,
487  .pre_divider = n,
488  .cpu_divider =
489  static_cast<uint8_t>(cpu_divider) };
490  }
491  } // cpu_divider loop
492  }
493  } // m for loop
494  } // n for loop
495  SJ2_ASSERT_FATAL(
496  false,
497  "Failed to calculate the PLL0 settings for the desired frequency.");
498  return PllSettings_t{ .multiplier = 0, .pre_divider = 0, .cpu_divider = 0 };
499  }

513  bool WaitForPllLockStatus(PllSelect pll) const
514  {
515  // Skip waiting for PLLSTAT register to update if running a host unit test
516  if constexpr (build::kPlatform == build::Platform::host)
517  {
518  return true;
519  }
520 
521  volatile uint32_t * pll_status_registers[] = {
522  &(system_controller->PLL0STAT), // NOLINT
523  &(system_controller->PLL1STAT)
524  };
525  const bit::Mask kLockStatusMasks[] = { MainPll::kLockStatus,
526  UsbPll::kLockStatus };
527  volatile uint32_t * status_register = pll_status_registers[Value(pll)];
528  const bit::Mask kLockStatusMask = kLockStatusMasks[Value(pll)];
529 
530  while (!bit::Read(*status_register, kLockStatusMask.position))
531  {
532  continue;
533  }
534 
535  return true;
536  }
