163  void DrawPixel(int32_t x, int32_t y, Color_t color) override
164  {
165  // The 3 least significant bits hold the bit position within the byte
166  uint32_t bit_position = y & 0b111;
167  // Each byte makes up a vertical column.
168  // Shifting by 3, which also divides by 8 (the 8-bits of a column), will
169  // be the row that we need to edit.
170  uint32_t row = y >> 3;
171  // Mask to clear the bit
172  uint32_t clear_mask = ~(1 << bit_position);
173  // Mask to set the bit, if color.alpha != 0
174  bool pixel_is_on = (color.alpha != 0);
175  uint32_t set_mask = pixel_is_on << bit_position;
176  // Address of the pixel column to edit
177  uint8_t * pixel_column = &(bitmap_[row][x]);
178  // Read pixel column and update the pixel
179  uint32_t result = (*pixel_column & clear_mask) | set_mask;
180  // Update pixel with the result of this operation
181  *pixel_column = static_cast<uint8_t>(result);
182  }
