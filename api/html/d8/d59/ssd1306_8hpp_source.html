98  void DrawPixel(int32_t x, int32_t y, Color_t color) override
99  {
100  // The 3 least significant bits hold the bit position within the byte
101  uint32_t bit_position = y & 0b111;
102 
103  // Each byte makes up a vertical column.
104  // Shifting by 3, which also divides by 8 (the 8-bits of a column), will
105  // be the row that we need to edit.
106  uint32_t row = y >> 3;
107 
108  // Mask to clear the bit
109  uint32_t clear_mask = ~(1 << bit_position);
110 
111  // Mask to set the bit, if color.alpha != 0
112  bool pixel_is_on = !color.IsBlank();
113  uint32_t set_mask = pixel_is_on << bit_position;
114 
115  // Address of the pixel column to edit
116  uint8_t * pixel_column = &(bitmap_[row][x]);
117 
118  // Read pixel column and update the pixel
119  uint32_t result = (*pixel_column & clear_mask) | set_mask;
120 
121  // Update pixel with the result of this operation
122  *pixel_column = static_cast<uint8_t>(result);
123  }

234  void Write(uint32_t data, Transaction transaction, size_t size = 1)
235  {
236  dc_.Set(static_cast<sjsu::Gpio::State>(transaction));
237  cs_.Set(sjsu::Gpio::State::kLow);
238  for (size_t i = 0; i < size; i++)
239  {
240  uint8_t send = static_cast<uint8_t>(data >> (((size - 1) - i) * 8));
241  if (transaction == Transaction::kCommand)
242  {
243  sjsu::LogDebug("send = 0x%X", send);
244  }
245  spi_.Transfer(send);
246  }
247  cs_.Set(sjsu::Gpio::State::kHigh);
248  }
