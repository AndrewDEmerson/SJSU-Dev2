173  static void I2cHandler(const PartialBus_t & i2c)
174  {
175  MasterState state = MasterState(i2c.registers->STAT);
176  uint32_t clear_mask = 0;
177  uint32_t set_mask = 0;
178  switch (state)
179  {
180  case MasterState::kBusError: // 0x00
181  {
182  i2c.transaction.status = Status::kBusError;
183  set_mask = Control::kAssertAcknowledge | Control::kStop;
184  break;
185  }
186  case MasterState::kStartCondition: // 0x08
187  {
188  i2c.registers->DAT = i2c.transaction.GetProperAddress();
189  break;
190  }
191  case MasterState::kRepeatedStart: // 0x10
192  {
193  i2c.transaction.operation = Operation::kRead;
194  i2c.registers->DAT = i2c.transaction.GetProperAddress();
195  break;
196  }
197  case MasterState::kSlaveAddressWriteSentRecievedAck: // 0x18
198  {
199  clear_mask = Control::kStart;
200  if (i2c.transaction.out_length == 0)
201  {
202  i2c.transaction.busy = false;
203  i2c.transaction.status = Status::kSuccess;
204  set_mask = Control::kStop;
205  }
206  else
207  {
208  size_t position = i2c.transaction.position++;
209  i2c.registers->DAT = i2c.transaction.data_out[position];
210  }
211  break;
212  }
213  case MasterState::kSlaveAddressWriteSentRecievedNack: // 0x20
214  {
215  clear_mask = Control::kStart;
216  i2c.transaction.busy = false;
217  i2c.transaction.status = Status::kDeviceNotFound;
218  set_mask = Control::kStop;
219  break;
220  }
221  case MasterState::kTransmittedDataRecievedAck: // 0x28
222  {
223  if (i2c.transaction.position >= i2c.transaction.out_length)
224  {
225  if (i2c.transaction.repeated)
226  {
227  // OR with 1 to set address as READ for the next transaction
228  i2c.transaction.operation = Operation::kRead;
229  i2c.transaction.position = 0;
230  set_mask = Control::kStart;
231  }
232  else
233  {
234  i2c.transaction.busy = false;
235  set_mask = Control::kStop;
236  }
237  }
238  else
239  {
240  size_t position = i2c.transaction.position++;
241  i2c.registers->DAT = i2c.transaction.data_out[position];
242  }
243  break;
244  }
245  case MasterState::kTransmittedDataRecievedNack: // 0x30
246  {
247  i2c.transaction.busy = false;
248  set_mask = Control::kStop;
249  break;
250  }
251  case MasterState::kArbitrationLost: // 0x38
252  {
253  set_mask = Control::kStart;
254  break;
255  }
256  case MasterState::kSlaveAddressReadSentRecievedAck: // 0x40
257  {
258  clear_mask = Control::kStart;
259  if (i2c.transaction.in_length == 0)
260  {
261  set_mask = Control::kStop;
262  }
263  // If we only want 1 byte, make sure to nack that byte
264  else if (i2c.transaction.in_length == 1)
265  {
266  clear_mask |= Control::kAssertAcknowledge;
267  }
268  // If we want more then 1 byte, make sure to ack the first byte
269  else
270  {
271  set_mask = Control::kAssertAcknowledge;
272  }
273  break;
274  }
275  case MasterState::kSlaveAddressReadSentRecievedNack: // 0x48
276  {
277  clear_mask = Control::kStart;
278  i2c.transaction.status = Status::kDeviceNotFound;
279  i2c.transaction.busy = false;
280  set_mask = Control::kStop;
281  break;
282  }
283  case MasterState::kRecievedDataRecievedAck: // 0x50
284  {
285  const size_t kBufferEnd = i2c.transaction.in_length;
286  if (i2c.transaction.position < kBufferEnd)
287  {
288  const size_t kPosition = i2c.transaction.position;
289  i2c.transaction.data_in[kPosition] =
290  static_cast<uint8_t>(i2c.registers->DAT);
291  i2c.transaction.position++;
292  }
293  // Check if the position has been pushed past the buffer length
294  if (i2c.transaction.position + 1 >= kBufferEnd)
295  {
296  clear_mask = Control::kAssertAcknowledge;
297  i2c.transaction.busy = false;
298  }
299  else
300  {
301  set_mask = Control::kAssertAcknowledge;
302  }
303  break;
304  }
305  case MasterState::kRecievedDataRecievedNack: // 0x58
306  {
307  i2c.transaction.busy = false;
308  if (i2c.transaction.in_length != 0)
309  {
310  size_t position = i2c.transaction.position++;
311  i2c.transaction.data_in[position] =
312  static_cast<uint8_t>(i2c.registers->DAT);
313  }
314  set_mask = Control::kStop;
315  break;
316  }
317  case MasterState::kDoNothing: // 0xF8
318  {
319  break;
320  }
321  default:
322  {
323  clear_mask = Control::kStop;
324  SJ2_ASSERT_FATAL(false, "Invalid I2C State Reached!!");
325  break;
326  }
327  }
328  // Clear I2C Interrupt flag
329  clear_mask |= Control::kInterrupt;
330  // Set register controls
331  i2c.registers->CONSET = set_mask;
332  i2c.registers->CONCLR = clear_mask;
333  }
