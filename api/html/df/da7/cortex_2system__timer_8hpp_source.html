95  // Set all flags required to enable the counter
96  uint32_t ctrl_mask = (1 << ControlBitMap::kTickInterupt) |
97  (1 << ControlBitMap::kEnableCounter) |
98  (1 << ControlBitMap::kClkSource);
99  // Set the system tick counter to start immediately
100  sys_tick->VAL = 0;
101  sys_tick->CTRL |= ctrl_mask;
102
