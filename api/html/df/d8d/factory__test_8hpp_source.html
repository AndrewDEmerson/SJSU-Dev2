203  int CalculateTemperature(I2c & temperature_i2c)
204  {
205  // Enables one burst mode (on-demand) temperature sampling
206  // by setting control register 0xC4 oneburst bit 3 to 1
207  // and stop bit 2 to 0. Stop bit is set to 1 upon successful
208  // measurement.
209  temperature_i2c.Write(kTemperatureAddress, { 0xC4, 0x04 });
210 
211  // Enable the auto increment of the i2c register address pointer
212  // by setting control register 0xC5 to autoinc bit to 1
213  temperature_i2c.Write(kTemperatureAddress, { 0xC5, 0x01 });
214  // MSB, LSB for temperature data
215  uint8_t temperature_ms = 0xC1;
216  uint8_t temperature_ls = 0xC2;
217  int temperature_data = 0;
218  temperature_i2c.WriteThenRead(kTemperatureAddress, &temperature_ms, 1,
219  &temperature_ms, 1);
220  temperature_i2c.Read(kTemperatureAddress, &temperature_ls, 1);
221 
222  temperature_data = ((temperature_ms & 0x7F) << 8) | temperature_ls;
223  printf(" Temperature Data: 0x%02X, 0x%02X, 0x%04X\n\n", temperature_ms,
224  temperature_ls, temperature_data);
225  // Compute the actual temperature in Celsius
226  return (55 + ((temperature_data - 16384) / 160));
227  }
