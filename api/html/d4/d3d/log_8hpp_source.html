67  snprintf(buffer.data(), buffer.size(),

76  bytes_left = std::max(static_cast<int>(buffer.size()) - position, 0);
77  position += snprintf(&buffer[position], bytes_left, format, params...);
78 
79  // Write the ending of the log statement that appends a newline and a color
80  // reset command.
81  bytes_left = std::max(static_cast<int>(buffer.size()) - position, 0);
82  position += snprintf(&buffer[position], bytes_left, "\n" SJ2_COLOR_RESET);

86  // trimmed/lost.

92  std::copy(kEllipsisEnding.begin(), kEllipsisEnding.end(),
93  &buffer[position]);
