69  // The EEPROM runs at 375 kHz
70  constexpr float kEepromClk = 375'000;
71  constexpr float kNanosecond = 1E-9f;

73  // The EEPROM is turned on by default, but in case it was somehow
74  // turned off, we turn it on by writing 0 to the PWRDWN register
75  eeprom_register->PWRDWN = 0;

77  // Initialize EEPROM wait state register with number of wait states
78  // for each of its internal phases
79  // Phase 3 (15 ns)
80  eeprom_register->WSTATE |=
81  static_cast<uint8_t>((15 * kNanosecond * kSystemClock) + 1);
82  // Phase 2 (55 ns)
83  eeprom_register->WSTATE |=

99  void Write(const uint8_t * wdata, uint32_t full_address,
100  size_t count) const override
101  {
102  constexpr bit::Mask kLower6Bits = bit::CreateMaskFromRange(0, 5);
103  constexpr bit::Mask kUpper6Bits = bit::CreateMaskFromRange(6, 11);
104 
105  full_address = bit::Clear(full_address, kAddressMask);
106 
107  // Because the EEPROM uses 32-bit communication, write_data will be casted
108  // into a uint32_t *
109  const uint32_t * write_data = reinterpret_cast<const uint32_t *>(wdata);
110 
111  // The first 6 bits in the address (MSB) dictate which page is being written
112  // to in the EEPROM, and the last 6 bits (LSB) dictate offset in the page
113  uint8_t page_count = bit::Read(full_address, kUpper6Bits);
114  uint8_t page_offset = bit::Read(full_address, kLower6Bits);
115 
116  uint16_t address;
117 
118  for (size_t i = 0; i*4 < count; i++)
119  {
120  // Page offset is incremented by 4 because we're writing 32 bits
121  constexpr uint8_t kOffsetInterval = 4;
122 
123  address = static_cast<uint16_t>((page_count << 6) + page_offset);
124 
125  eeprom_register->ADDR = address;
126  eeprom_register->CMD = kWrite32Bits;
127  eeprom_register->WDATA = write_data[i];
128 
129  // Poll status register bit to see when writing is finished
130  auto check_register = [] ()
131  {
132  return !(bit::Read(eeprom_register->INT_STATUS,
133  Status::kReadWriteStatusMask));
134  };
135 
136  Wait(kMaxTimeout, check_register);
137 
138  // Clear write interrupt
139  eeprom_register->INT_CLR_STATUS =
140  (bit::Set(0, Status::kReadWriteStatusMask));
141 
142  page_offset = static_cast<uint8_t>(page_offset + kOffsetInterval);
143 
144  // If the 64 byte page buffer fills up, then it must be programmed to the
145  // programmed to the EEPROM before continuing.
146  if (page_offset > 64)
147  {
148  Program(address);
149  page_count++;

152  }
153

155  Program(address);
