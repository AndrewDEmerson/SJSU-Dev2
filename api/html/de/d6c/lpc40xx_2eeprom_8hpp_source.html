44  // Initializing the EEPROM requires setting the wait state register,
45  // and the clock divider register
46  void Initialize() const override
47  {
48  const float kSystemClock = static_cast<float>
49  (system_controller_.GetSystemFrequency());
50  constexpr float kEepromClk = 375'000; // EEPROM runs at 375 kHz
51  constexpr float kNanosecond = 1E-9f;
52 
53  // The EEPROM is turned on by default, but in case it was somehow
54  // turned off, we turn it on by writing 0 to the PWRDWN register
55  eeprom_register->PWRDWN = 0;
56 
57  // Initialzie EEPROM wait state register with number of wait states
58  // for each of its internal phases
59  // Phase 3 (15 ns)
60  eeprom_register->WSTATE |=
61  static_cast<uint8_t>((15 * kNanosecond * kSystemClock) + 1);
62  // Phase 2 (55 ns)
63  eeprom_register->WSTATE |=
64  (static_cast<uint8_t>((55 * kNanosecond * kSystemClock) + 1)) << 8;
65  // Phase 1 (35 ns)
66  eeprom_register->WSTATE |=
67  (static_cast<uint8_t>((35 * kNanosecond * kSystemClock) + 1)) << 16;
68 
69  // Initialize EEPROM clock
70  eeprom_register->CLKDIV = static_cast<uint8_t>(kSystemClock / kEepromClk);
71  }

73  // The EEPROM can only be interfaced in 32-bit mode, which means that the
74  // address used must be a multiple of 4. The address entered will be cut
75  // down to a multiple of 4.
76  void Write(const uint8_t * wdata, uint32_t full_address,
77  size_t count) const override
78  {
79  constexpr uint8_t kMax6Bits = 0b11'1111;
80 
81  // Writing a 32-bit number to an address that isn't a multiple of 4 will
82  // cause a hard fault, thus this safety is put into place to avoid that.
83  full_address &= kAddressMask;
84 
85  // Because the EEPROM uses 32-bit communication, write_data will be casted
86  // into a uint32_t *
87  const uint32_t * write_data = reinterpret_cast<const uint32_t *>(wdata);
88 
89  // The first 6 bits in the address (MSB) dictate which page is being written
90  // to in the EEPROM, and the last 6 bits (LSB) dictate offset in the page
91  uint8_t page_count = static_cast<uint8_t>(full_address >> 6);
92  uint8_t page_offset = full_address & 0b0000'0011'1111;
93 
94  uint16_t address;
95 
96  for (size_t i = 0; i*4 < count; i++)
97  {
98  // Page offset is incremented by 4 because we're writing 32 bits
99  constexpr uint8_t kOffsetInterval = 4;
100 
101  address = static_cast<uint16_t>((page_count << 6) + page_offset);
102 
103  eeprom_register->ADDR = address;
104  eeprom_register->CMD = kWrite32Bits;
105  eeprom_register->WDATA = write_data[i];
106 
107  // Poll status register bit to see when writing is finished
108  auto check_register = [&] ()
109  {
110  return !(eeprom_register->INT_STATUS >> kReadWriteStatusBit & 1);
111  };
112 
113  Wait(std::chrono::milliseconds(kMaxTimeout), check_register);
114 
115  // Clear write interrupt
116  eeprom_register->INT_CLR_STATUS = (1 << kReadWriteStatusBit);
117 
118  page_offset = static_cast<uint8_t>(page_offset + kOffsetInterval);
119 
120  // If the 64 byte page buffer fills up, then it must be programmed to the
121  // EEPROM before continuing.
122  if (page_offset > kMax6Bits)
123  {
124  Program(address);
125  page_count++;
126  page_offset = 0;
127  }
128  }
129 
130  // Program final information so that it's stored in the EEPROM
131  Program(address);
132  }

134  // The EEPROM is accessed through a 64 byte page buffer, and after it fills
135  // up, it must be programmed to the actual EEPROM. This function handles that.
136  void Program(uint32_t address) const
137  {
138  eeprom_register->ADDR = address;
139  eeprom_register->CMD = kEraseProgram;
140 
141  // Poll status register bit to see when writing is finished
142  auto check_register = [&] ()
143  {
144  return !(eeprom_register->INT_STATUS >> kProgramStatusBit & 1);
145  };
146 
147  // Wait((!(eeprom_register->INT_STATUS >> kReadWriteStatusBit)) & 1);
148  Wait(std::chrono::milliseconds(kMaxTimeout), check_register);
149 
150  // Clear program interrupt
151  eeprom_register->INT_CLR_STATUS = (1 << kProgramStatusBit);
152  }

154  // This function will return however much 32-bit data from the EEPROM
155  // starting at address and continuing on for how large count is. The EEPROM
156  // can only be interfaced in 32-bit mode, which means that the address used
157  // must be a multiple of 4. The address entered will be cut down to a
