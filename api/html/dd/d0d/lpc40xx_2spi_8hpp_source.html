187 
188  // Power up peripheral

190  bus_.power_on_bit);
191  // Enable SSP pins
192  bus_.mosi.SetPinFunction(bus_.pin_function_id);
193  bus_.miso.SetPinFunction(bus_.pin_function_id);
194  bus_.sck.SetPinFunction(bus_.pin_function_id);
195  // Set SSP frame format to SPI
196  bus_.registers->CR0 = bit::Insert(
197  bus_.registers->CR0, kSpiFormatCode, ControlRegister0::kFrameBit);
198  // Set SPI to master mode by clearing
199  bus_.registers->CR1 =

264  void SetClock(units::frequency::hertz_t frequency,
265  bool positive_clock_on_idle = false,

267  {
268  bus_.registers->CR0 = bit::Insert(bus_.registers->CR0,

274 
275  auto system_frequency =
276  sjsu::SystemController::GetPlatformController().GetPeripheralFrequency(
277  bus_.power_on_bit);
278 
279  uint16_t prescaler = (system_frequency / frequency).to<uint16_t>();
280  // Store lower half of prescalar in clock prescalar register
281  bus_.registers->CPSR = prescaler & 0xFF;
282  // Store upper 8 bit half of the prescalar in control register 0
283  bus_.registers->CR0 = bit::Insert(
284  bus_.registers->CR0, prescaler >> 8, ControlRegister0::kDividerBit);
285  }
