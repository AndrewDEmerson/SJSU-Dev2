103  enum class Frequency : uint8_t
104  {
105  k1Hz = 0, // NOLINT(readability-identifier-naming)
106  k10Hz, // NOLINT(readability-identifier-naming)
107  k100Hz, // NOLINT(readability-identifier-naming)
108  k1000Hz, // NOLINT(readability-identifier-naming)
109  kCount
110  };
111 
112  static constexpr size_t kMaxTaskCount = Value(Frequency::kCount);
113 
114  PeriodicScheduler()
115  : Task("Periodic_Scheduler", Priority::kLow), task_list_{ nullptr }
116  {
117  const uint32_t kPeriods[] = { 1000, 100, 10, 1 };
118  const char * timer_names[] = { "Timer_1Hz", "Timer_10Hz", "Timer_100Hz",
119  "Timer_1000Hz" };
120  for (uint8_t i = 0; i < kMaxTaskCount; i++)
121  {
122  timers_[i] =
123  xTimerCreateStatic(timer_names[i],
124  kPeriods[i], // timer period in ticks
125  pdTRUE, // auto-reload
126  nullptr, // pvTimerID
127  HandlePeriodicTimer, // pxCallbackFunction
128  &(timer_buffers_[i]));
129  SJ2_ASSERT_FATAL(timers_[i] != nullptr,
130  "Failed to create timer for: %s\n", timer_names[i]);
131  SJ2_ASSERT_FATAL(xTimerStart(timers_[i], 0) == pdPASS,
132  "Failed to set timer into the active state for: %s\n",
133  timer_names[i]);
134  }
135  }
136  bool Run() override
137  {
138  return true;
139  }
144  void SetTask(PeriodicTaskInterface * task, Frequency frequency)
145  {
146  task_list_[Value(frequency)] = task;
147  // Since HandlePeriodicTimer callback handles multiple timers, the pointer
148  // of the task is stored as the timer's pvTimerID in order to be able to
149  // identify which task should be handled in the HandlePeriodicTimer()
150  // function.
151  vTimerSetTimerID(timers_[Value(frequency)],
152  static_cast<void *>(task));
153  }
156  PeriodicTaskInterface * GetTask(Frequency frequency) const
157  {
158  return task_list_[Value(frequency)];
159  }

167  static void HandlePeriodicTimer(TimerHandle_t timer)
168  {
169  PeriodicTaskInterface * task =
170  static_cast<PeriodicTaskInterface *>(pvTimerGetTimerID(timer));
171  if (task != nullptr)
172  {
173  xSemaphoreGive(task->GetPeriodicSemaphore());
