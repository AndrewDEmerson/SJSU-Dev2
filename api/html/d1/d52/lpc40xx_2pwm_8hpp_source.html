184  Status Initialize(units::frequency::hertz_t frequency_hz) const override
185  {
186  SJ2_ASSERT_FATAL(1 <= channel_.channel && channel_.channel <= 6,
187  "Channel must be between 1 and 6 on LPC40xx platforms.");
188 
189  system_controller_.PowerUpPeripheral(channel_.peripheral.id);
190 
191  auto * pwm = channel_.peripheral.registers;
192  // Set prescalar to 1 so the input frequency to the PWM peripheral is equal
193  // to the peripheral clock frequency.
194  //
195  // PR = Prescale register = defines the maximum value for the prescaler
196  pwm->PR = 0;
197 
198  // PC = Prescale counter. Set to 0 to cause the timer counter to increment
199  // after each peripheral clock tick.
200  pwm->PC = 0;
201 
202  // Mode 0x0 means increment time counter (TC) after the peripheral clock
203  // has cycled the amount inside of the prescale.
204  pwm->CTCR = bit::Insert(pwm->CTCR, 0, CountControl::kMode);
205 
206  // 0x0 for this register says to use the TC for input counts.
207  pwm->CTCR = bit::Insert(pwm->CTCR, 0, CountControl::kCountInput);
208 
209  // Match register 0 is used to generate the desired frequency. If the time
210  // counter TC is equal to MR0
211  const units::frequency::hertz_t kPeripheralFrequency =
212  system_controller_.GetPeripheralFrequency(channel_.peripheral.id);
213 
214  pwm->MR0 = (kPeripheralFrequency / frequency_hz).to<uint32_t>();
215 
216  // Sets match register 0 to reset when TC and Match 0 match each other,
217  // meaning that the PWM pulse will cycle continuously.
218  pwm->MCR = bit::Set(pwm->MCR, MatchControl::kPwm0Reset);
219 
220  // Enables PWM TC and PC for counting and enables PWM mode
221  EnablePwm();
222 
223  // Enables PWM[channel] output
224  uint32_t channel_bit_position =
225  OutputControl::kEnableOutput.position + channel_.channel;
226  pwm->PCR = bit::Set(pwm->PCR, channel_bit_position);
227 
228  channel_.pin.SetPinFunction(channel_.pin_function_code);
229 
230  return Status::kSuccess;
231  }

233  void SetDutyCycle(float duty_cycle) const override
234  {
235  SJ2_ASSERT_FATAL(0.0f <= duty_cycle && duty_cycle <= 1.0f,
236  "duty_cycle of Duty Cycle provided is out of bounds.");
237  *GetMatchRegisters(channel_.channel) = CalculateDutyCycle(duty_cycle);
238  // LER (Load Enable Register) sets the hardware up to load the new duty
239  // cycle period in the next reset cycle after MR0 match. Hardware requires
240  // this as a means ot prevent itself from glitching and changing pwm
241  // instantaniously.
242  channel_.peripheral.registers->LER |= (1 << channel_.channel);
243  }

252  void SetFrequency(units::frequency::hertz_t frequency_hz) const override
253  {
254  SJ2_RETURN_IF(
255  frequency_hz == 0_Hz,
256  "Cannot set frequency to 0Hz! This call will have no effect!");
257 
258  auto peripheral_frequency =
259  system_controller_.GetPeripheralFrequency(channel_.peripheral.id);
260  // Get the current duty cycle so we can match it to the updated frequency.
261  float previous_duty_cycle = GetDutyCycle();
262 
263  // In order to avoid PWM glitches, the PWM must be disabled while updating
264  // the MR0 register. Doing this will reset all counters to 0 and allow us to
265  // update MR0.
266  EnablePwm(false);
267  {
268  auto new_frequency = (peripheral_frequency / frequency_hz).to<uint32_t>();
269  *GetMatchRegisters(0) = new_frequency;
270  SetDutyCycle(previous_duty_cycle);
271  }
272  EnablePwm(true);
273  }

297  void EnablePwm(bool enable = true) const
298  {
299  volatile uint32_t * pwm_timer = &channel_.peripheral.registers->TCR;
300  if (enable)
301  {
302  // Reset the Timer Counter

304  // Clear reset and allow timer to count

306  // Enable PWM output

308  // Enable counting
309  *pwm_timer = bit::Set(*pwm_timer, Timer::kCounterEnable);
310  }

313  // Disable PWM output
314  *pwm_timer = bit::Clear(*pwm_timer, Timer::kPwmEnable);
