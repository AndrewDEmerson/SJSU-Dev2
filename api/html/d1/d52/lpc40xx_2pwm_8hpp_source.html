134  Status Initialize(units::frequency::hertz_t frequency_hz) const override
135  {
136  SJ2_ASSERT_FATAL(1 <= channel_.channel && channel_.channel <= 6,
137  "Channel must be between 1 and 6 on LPC40xx platforms.");
138 
139  system_controller_.PowerUpPeripheral(channel_.peripheral.power_on_id);
140  // Set prescalar to 1 so the input frequency to the PWM peripheral is equal
141  // to the peripheral clock frequency.
142  //
143  // PR = Prescale register = defines the maximum value for the prescaler
144  channel_.peripheral.registers->PR = 0;
145  // PC = Prescale counter. Set to 0 to cause the timer counter to increment
146  // after each peripheral clock tick.
147  channel_.peripheral.registers->PC = 0;
148  // Mode 0x0 means increment time counter (TC) after the peripheral clock
149  // has cycled the amount inside of the prescale.
150  channel_.peripheral.registers->CTCR = bit::Insert(
151  channel_.peripheral.registers->CTCR, 0, CountControl::kMode);
152  // 0x0 for this register says to use the TC for input counts.
153  channel_.peripheral.registers->CTCR = bit::Insert(
154  channel_.peripheral.registers->CTCR, 0, CountControl::kCountInput);
155  // Match register 0 is used to generate the desired frequency. If the time
156  // counter TC is equal to MR0
157  const units::frequency::hertz_t kPeripheralFrequency =
158  system_controller_.GetPeripheralFrequency(
159  channel_.peripheral.power_on_id);
160  channel_.peripheral.registers->MR0 =
161  (kPeripheralFrequency / frequency_hz).to<uint32_t>();
162  // Sets match register 0 to reset when TC and Match 0 match each other,
163  // meaning that the PWM pulse will cycle continuously.
164  channel_.peripheral.registers->MCR = bit::Set(
165  channel_.peripheral.registers->MCR, MatchControl::kPwm0Reset.position);
166  // Enables PWM TC and PC for counting and enables PWM mode
167  EnablePwm();
168  // Enables PWM[channel] output
169  channel_.peripheral.registers->PCR =
170  bit::Set(channel_.peripheral.registers->PCR,
171  OutputControl::kEnableOutput.position + channel_.channel);
172 
173  channel_.pin.SetPinFunction(channel_.pin_function_id);
174 
175  return Status::kSuccess;
176  }

192  void SetFrequency(units::frequency::hertz_t frequency_hz) const override
193  {
194  SJ2_ASSERT_FATAL(frequency_hz != 0_Hz, "Pwm Frequency cannot be zero Hz.");
195  // Disables PWM mode; this will reset all counters to 0

222  void EnablePwm(bool enable = true) const
223  {
224  volatile uint32_t * pwm_timer = &channel_.peripheral.registers->TCR;
225  if (enable)
226  {
227  // Reset the Timer Counter
228  *pwm_timer = bit::Set(*pwm_timer, Timer::kCounterReset.position);
229  // Clear reset and allow timer to count
230  *pwm_timer = bit::Clear(*pwm_timer, Timer::kCounterReset.position);
231  // Enable PWM output
232  *pwm_timer = bit::Set(*pwm_timer, Timer::kPwmEnable.position);
233  // Enable counting
234  *pwm_timer = bit::Set(*pwm_timer, Timer::kCounterEnable.position);
235  }
236  else
237  {
238  // Disable PWM output
239  *pwm_timer = bit::Clear(*pwm_timer, Timer::kPwmEnable.position);
240  }
241  }
